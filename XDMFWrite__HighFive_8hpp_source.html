676 inline std::string join(const std::vector<std::string>& lines, const std::string& sep)
677 {
678  if (lines.size() == 1) {
679  return lines[0];
680  }
681 
682  std::string ret = "";
683 
684  for (auto line : lines) {
685  if (ret.size() == 0) {
686  ret += line;
687  continue;
688  }
689 
690  if (line[0] == sep[0]) {
691  ret += line;
692  }
693  else if (ret[ret.size() - 1] == sep[0]) {
694  ret += line;
695  }
696  else {
697  ret += sep + line;
698  }
699  }
700 
701  return ret;
702 }
